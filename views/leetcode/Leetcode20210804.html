    # 1.返回倒数第K个节点

    # 要求

    实现一种算法，找出单向链表中倒数第K个节点。返回该节点的值

    # 实例1
    输入： 1->2->3->4->5 和 k = 2
输出： 4
    1
    2

    # 解题方法

    # 本人方法

    我使用了双指针的方法，先让指针1走k步，再让两个指针同时走直到指针1到达尾端，那么此时指针2指向的就是输出的值

    var kthToLast = function(head, k) {
    if(!head) return head
    let p1 = head
    let p2 = head
    while( k>0 ){
        p1 = p1.next
        k--
    }
    while(p1){
        p1 = p1.next
        p2 = p2.next
    }
    return p2.val
};

    # 2.删除中间节点

    # 要求

    若链表中的某个节点，既不是链表头节点，也不是链表尾节点，则称其为该链表的「中间节点」。

    假定已知链表的某一个中间节点，请实现一种算法，将该节点从链表中删除。

    例如，传入节点 c（位于单向链表 a->b->c->d->e->f 中），将其删除后，剩余链表为 a->b->d->e->f

    # 实例1
    输入：节点 5 （位于单向链表 4->5->1->9 中）
输出：不返回任何数据，从链表中删除传入的节点 5，使链表变为 4->1->9

    # 3.分割链表

    # 要求

    给你一个链表的头节点 head 和一个特定值 x ，请你对链表进行分隔，使得所有 小于 x 的节点都出现在 大于或等于 x 的节点之前。

    你不需要 保留 每个分区中各节点的初始相对位置。

    # 实例1
    输入：head = [1,4,3,2,5,2], x = 3
输出：[1,2,2,4,3,5]
    1
    2
    # 实例2
    输入：head = [2,1], x = 2
输出：[1,2]
    1
    2

    # 解题方法

    # 本人方法

    通过创建两个新链表 一个存放比x小 另一个反之 再连起来

    var partition = function(head, x) {

    if(!head) return null

    let small = new ListNode(0);  //两个空指针为首的链表
    const smallHead = small;

    let large = new ListNode(0);
    const largeHead = large;

    while (head !== null) {
        if (head.val < x) {
            small.next = head;
            small = small.next;
        } else {
            large.next = head;
            large = large.next;
        }
        head = head.next;
    }

    large.next = null;
    small.next = largeHead.next; //链接两个链表 去掉空指针
    return smallHead.next;
};

    # 4.链表求和

    # 要求

    给定两个用链表表示的整数，每个节点包含一个数位。

    这些数位是反向存放的，也就是个位排在链表首部。

    编写函数对这两个整数求和，并用链表形式返回结果。

    # 实例1
    输入：(7 -> 1 -> 6) + (5 -> 9 -> 2)，即617 + 295
输出：2 -> 1 -> 9，即912
    1
    2

    **进阶：**思考一下，假设这些数位是正向存放的，又该如何解决呢?

    # 进阶实例2
    输入：(6 -> 1 -> 7) + (2 -> 9 -> 5)，即617 + 295
输出：9 -> 1 -> 2，即912
    1
    2

    # 解题方法

    通过利用三目运算符 和进位来求和 不存在的位数当作0

    var addTwoNumbers = function (l1, l2) {
    let result = new ListNode(0)
    let p = result
    let p1 = l1
    let p2 = l2

    let carry = 0
    while (p1 || p2) {
        let p1Val = p1 ? p1.val : 0
        let p2Val = p2 ? p2.val : 0
        let sum = p1Val + p2Val + carry
        carry = Math.floor(sum / 10)
        p.next = new ListNode(sum % 10)
        p = p.next
        if (p1) p1 = p1.next
        if (p2) p2 = p2.next
    }

    if (carry) p.next = new ListNode(carry)
    return result.next
}

    # 5.回文链表

    # 要求

    编写一个函数，检查输入的链表是否是回文的。

    # 实例1
    输入： 1->2
输出： false
    1
    2
    # 实例2
    输入： 1->2->2->1
输出： true
    1
    2

    # 解题方法

    # 本人方法

    利用数组来进行回文判定

    var isPalindrome = function(head) {
    const vals = [];
    while (head !== null) {
        vals.push(head.val);
        head = head.next;
    }
    for (let i = 0, j = vals.length - 1; i < j; ++i, --j) {
        if (vals[i] !== vals[j]) {
            return false;
        }
    }
    return true;
};

    # 6.链表相交

    # 要求

    给你两个单链表的头节点 headA 和 headB ，请你找出并返回两个单链表相交的起始节点。如果两个链表没有交点，返回 null 。

    # 解题方法

    巧妙的利用拼接两个链表形如[AB] [BA]，容易得到若有相同节点，那么会在该节点指针相等

    var getIntersectionNode = function(headA, headB) {
    let A = headA
    let B = headB
    while(A !== B){
        A = A !== null ? A.next : headB
        B = B !== null ? B.next : headA
    }
    return A
};

    # 7.环路检测

    # 要求

    给定一个链表，如果它是有环链表，实现一个算法返回环路的开头节点。

    # 解题方法

    # 本人方法

    主要使用的是快慢指针，先找到相遇点，再通过相遇点与头部点一步一步移动，这两个指针再次相遇的点就是开头节点。（结论由数学计算获得）从相遇点到入环点的距离加上 n-1*n*−1 圈的环长，恰好等于从链表头部到入环点的距离。

    var detectCycle = function(head) {

    if(!head) return null

    let slow = head
    let fast = head

    while(fast != null){
        slow = slow.next
        if(fast.next != null){
            fast = fast.next.next
        }else{
            return null
        }
        if(slow == fast) {
            let p1 = head
            let p2 = slow

            while(p1 != p2){
            p1 = p1.next
            p2 = p2.next
           }
           return  p1
        }
    }
    return null
};

    # 大佬方法

    采用哈希表的方法，我们遍历链表中的每个节点，并将它记录下来；一旦遇到了此前遍历过的节点，就可以判定链表中存在环。借助哈希表可以很方便地实现。

    var detectCycle = function(head) {
    const visited = new Set();
    while (head !== null) {
        if (visited.has(head)) {
            return head;
        }
        visited.add(head);
        head = head.next;
    }
    return null;
};
