    # 1.判定是否为回文的排列

    # 要求

    给定一个字符串，编写一个函数判定其是否为某个回文串的排列之一。

    回文串是指正反两个方向都一样的单词或短语。排列是指字母的重新排列。

    回文串不一定是字典当中的单词。

    # 实例1
    输入："tactcoa"
输出：true（排列有"tacocat"、"atcocta"，等等）
    1
    2

    # 解题方法

    # 本人方法

    从回文的特征入手，可以考虑到回文只会最多存在一个出现次数为奇数的字母，所有我这里利用到了对象，通过key：value的形式来遍历value，看value为奇数的个数，若个数大于1则不是回文排列

    /**
 * @param {string} s
 * @return {boolean}
 */
var canPermutePalindrome = function(s) {
    let obj = {}
    let count = 0
    for (let i of s) {
        if( !obj[i] ){
            obj[i] = 1
        }else{
            obj[i]++
        }
    }
    for (let key in obj ){
        if( obj[key] %2 == 1) count++
    }
    if(count>1) {return false}
    else {return true}
};

    # 大佬方法

    1.set

    通过利用Set的成员唯一的结构，循环判断set中是否已经存在当前元素，若有则删除，无则添加，那么set中存在的只会是出现过奇数次数的元素，那么set.size就是出现次数为奇数的元素的个数

    /**
 * @param {string} s
 * @return {boolean}
 */
var canPermutePalindrome = function(s) {
    let a=new Set();
    for(let i of s){
        a.has(i)?a.delete(i):a.add(i)
    }
    return a.size<=1
};

    # 2.一次编辑

    # 要求

    字符串有三种编辑操作:插入一个字符、删除一个字符或者替换一个字符。 给定两个字符串，编写一个函数判定它们是否只需要一次(或者零次)编辑。

    # 实例1
    输入: 
first = "pale"
second = "ple"
输出: True
    1
    2
    3
    4
    # 实例2
    输入: 
first = "pales"
second = "pal"
输出: False
    1
    2
    3
    4

    # 解题方法

    要详细分析三种情况：

    第一种就是两者长度大于1为false

    第二种就是两者长度相同，那么就得对比两个字符串是否有字符上的区别且区别小于1

    第三种就是两者长度相差一，（长字符串在短字符串上插入了一个字符/反之删去），只需要在某个位置两个字符不相同的情况下比较长字符串的下一个字符是否与短字符串该字符相同即可

    /**
 * @param {string} first
 * @param {string} second
 * @return {boolean}
 */
var oneEditAway = function(first, second) {
    let len1 = first.length
    let len2 = second.length
    let count = 0
    if( Math.abs( len1 - len2 ) > 1 ) return false
    if( len1 == len2 ){
        for(let i = 0; i < len1; i++){
            if( first[i] != second[i] ) count++
            if( count > 1 ) return false
        }
    }else{
        let [long,short] = len1 > len2 ? [first,second] : [second,first]
        for(let i = 0; i < long.length; i++ ){
             if (long[i] !== short[i]) {
                return long.slice(i+1) === short.slice(i)
            }
        }
    }
    return true
};

    # 3.字符串压缩

    # 要求

    字符串压缩。利用字符重复出现的次数，编写一种方法，实现基本的字符串压缩功能。比如，字符串aabcccccaaa会变为a2b1c5a3。若“压缩”后的字符串没有变短，则返回原先的字符串。你可以假设字符串中只包含大小写英文字母（a至z）。

    # 实例1
    输入："aabcccccaaa"
输出："a2b1c5a3"
    1
    2
    # 实例2
    输入："abbccd"
输出："abbccd"
解释："abbccd"压缩后为"a1b2c2d1"，比原字符串长度更长。

    # 解题方法

    # 本人方法

    简单通过转化成数组进行遍历，判断下一个字符是否等于这个字符，通过位置来计算个数，最后再比较压缩与非压缩的长度

    /**
 * @param {string} S
 * @return {string}
 */
var compressString = function(S) {
    let a = S.split('')
	let pos = 0
	let str = ""
	console.log(a)
	for(let i=0; i < a.length; i++){
		if(a[i] != a[i+1]) {
			str = str + a[i] + (i - pos + 1)
			pos = i+1
		}
	}
	if(str.length < S.length){
		return str
	}else{
		return S
	}
};

    /**
 * @param {number[][]} matrix
 * @return {void} Do not return anything, modify matrix in-place instead.
 */
var rotate = function(matrix) {
    //先将矩阵进行转置
    for(let i = 0; i<matrix.length; i++){
        for(let j =i+1; j<matrix[i].length; j++){
            let swap = matrix[i][j];
            matrix[i][j] = matrix[j][i]
            matrix[j][i] = swap
        }
    }
    //利用函数将矩阵反转
    for(let i = 0; i<matrix.length; i++){
        matrix[i].reverse()
    }
    return matrix
};

    # 解题方法

    # 本人方法

    利用set结构记录要清零的列，在遍历每行时当场将改行变为0

    var setZeroes = function(matrix) {
        let b = new Set()
        let zero = false
		//找出矩阵0元素，记录列
        for(let i = 0; i<matrix.length; i++){
            for(let j = 0; j< matrix[i].length; j++){
                if (matrix[i][j] == 0){
                    zero = true
                    b.add(j)
                } 
            }
            if(zero == true){
                for(let k = 0; k< matrix[i].length; k++){
					matrix[i][k] = 0
				}
				zero = false
            }
        }
		//将列变为0
		for(let j of b){
			for(let i = 0; i<matrix.length; i++){
				matrix[i][j] = 0
			}
		}
		return matrix
    };

    var setZeroes = function(matrix) {
    let cache = []
    let r = matrix.length, c = matrix[0].length
    if (r < 1) return
    for (let i = 0; i < r; i++) {
        for (let j = 0; j < c; j++) {
            if (matrix[i][j] === 0) {
                cache.push([i, j])
            }
        }
    }
    for (let item of cache) {
        for (let i = 0; i < r; i++) {
            matrix[i][item[1]] = 0
        }
        for (let j = 0; j < c; j++) {
            matrix[item[0]][j] = 0
        }
    }
};

    var isFlipedString = function(s1, s2) {
    if(s1.length != s2.length){
        return false
    }
    if(s1 == s2){
        return true
    }
    let len = s1.length
    let x = ''
    let y = ''
    for(let i = 1; i <= len; i ++){
        x = s1.substr(0,i)
        y = s1.substr(i,len - i)
        if( y+x == s2){
            return true
        }
    }
    return false
};

    # 7.移除重复节点

    # 要求

    编写代码，移除未排序链表中的重复节点。保留最开始出现的节点。

    # 实例1
    输入：[1, 2, 3, 3, 2, 1]
输出：[1, 2, 3]
    1
    2
    # 实例2
    输入：[1, 1, 1, 1, 2]
输出：[1, 2]
    1
    2

    # 解题方法

    详细说明见注释

    /**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var removeDuplicateNodes = function(head) {
    if(head === null) return null  //先判断链表是否为空
    let s = new Set()     //定义一个set
    let front = head      //front表示当前节点
    let last = head.next  //last表示下一个节点
    s.add(front.val)      //将头节点存入set
    while(last){ 
        if(s.has(last.val)){   //如果下一个节点已经存在set中，则删去下一个节点，
            front.next = last.next  //当前节点的下一个节点为下下个节点
            last.next = null        //断开下一个节点
            last = front.next       //重新定义last
        }else{
            s.add(last.val)     //如果不存在则加入到set中
            front = front.next  //跳到下一个节点
            last = last.next
        }
    }
    return head
};
